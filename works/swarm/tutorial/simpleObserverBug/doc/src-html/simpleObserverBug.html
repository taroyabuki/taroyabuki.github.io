001    import swarm.*;
002    
003    /**
004     * モデル＋GUI<BR>
005     * <BR>
006     * シミュレーションのモデルとそれを表示するためのインターフェースを併せ持つ
007     * ObserverSwarmを作る。
008     * これによってシステムは、ObserverSwarmがModelSwarmを作り、
009     * ModelSwarmがBugやFoodSpaceを作るという階層構造になる。
010     * このような階層構造ができると、シミュレーションも階層的に実行することになる。<BR>
011     * <BR>
012     * ここで用いるswarmはこれまでと違い、GUISwarmである。
013     * これによって、swarmはユーザとGUIを用いて対話することができる。
014     * とはいっても、生成の仕方などが大きく変わるわけではない。<BR>
015     * <BR>
016     * ひとつ変わったのは、シミュレーションを開始するのに、swarmのactivityを
017     * 取得してrunさせる必要がなくなったことである。これに相当するのは、ControlPanelが
018     * startボタンを押されたときに行う。<BR>
019     * <BR>
020     * <IMG src="../simpleObserverBug.png" border="0"><BR>
021     * <BR>
022     * 次はsimpleObserverBug2
023     * @author YABUKI Taro
024     * @version 0.4
025     */
026    public class simpleObserverBug{
027            public static void main(String[] args) {
028                    ObserverSwarm observerSwarm;
029                    Globals.env.initSwarm("bug", "0.4", "YABUKI Taro", args);
030                    
031                    observerSwarm=new ObserverSwarm(Globals.env.globalZone);
032                    observerSwarm.buildObjects();
033                    observerSwarm.buildActions();
034                    observerSwarm.activateIn(null);
035                    
036                    // We tell the swarm itself to go, instead of an activity
037                    // because the observerSwarm is a GUI-swarm, and has its
038                    // own controlPanel that we can talk to.
039                    observerSwarm.go();
040            }
041    }
