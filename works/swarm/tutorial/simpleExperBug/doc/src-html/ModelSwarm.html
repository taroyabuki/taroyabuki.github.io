001    import swarm.*;

008    /**
009     * ExperSwarm（正確にはParameterManager）からモデルのパラメータを設定できるように、
010     * メソッドをいくつか追加した。
011     * さらに、餌がいつ食べ尽くされたかを調べるために、時間を計るようにした。
012     */
013    public class ModelSwarm extends SwarmImpl{
014            public int worldXSize, worldYSize;
015            
016            public double seedProb;
017            public double bugDensity;
018            
019            int time;
020            
021            FoodSpace foodSpace;
022            Grid2d world;
023            
024            List bugList;
025            ActionGroup modelActions;
026            Schedule modelSchedule;
027            
028            public List bugList(){
029                    return bugList;
030            }
031            
032            public Grid2d getWorld(){
033                    return world;
034            }
035            
036            public FoodSpace getFoodSpace(){
037                    return foodSpace;
038            }
039    
040            /**
041             * @return タイムステップ数
042             */
043            public int getTime(){
044                    return time;
045            }
046            
047            public Object setWorldXSize$YSize(int x,int y){
048                    worldXSize=x;
049                    worldYSize=y;
050                    return this;
051            }
052            
053            /**
054             * ExperSwarmからパラメータを設定するためのメソッド
055             * @param s 餌の密度seedProb
056             * @param b Bugの密度bugDensity
057             */
058            public Object setSeedProb$bugDensity(double s,double b){
059                    seedProb=s;
060                    bugDensity=b;
061                    return this;
062            }
063            
064            public ModelSwarm(Zone aZone){
065                    super(aZone);
066                    
067                    worldXSize = 80;
068                    worldYSize = 80;
069                    seedProb   = 0.5;
070                    bugDensity = 0.1;
071            }
072            
073            public Object buildObjects(){
074                    Bug aBug;
075                    int x,y;
076                    
077                    foodSpace=new FoodSpace(this,worldXSize,worldYSize);
078                    foodSpace.seedFoodWithProb(seedProb);
079                    
080                    world=new Grid2dImpl(this,worldXSize,worldYSize);
081                    world.fillWithObject(null);
082                    
083                    bugList=new ListImpl(this);
084                    
085                    for (y = 0; y < worldYSize; y++){
086                            for (x = 0; x < worldXSize; x++){
087                                    if (Globals.env.uniformDblRand.getDoubleWithMin$withMax(0.0,1.0) < bugDensity){
088                                            aBug=new Bug(this);
089                                            aBug.setWorld$Food(world,foodSpace);
090                                            aBug.setX$Y(x,y);
091                                            bugList.addLast(aBug);
092                                    }
093                            }
094                    }
095                    
096                    time=0;
097                    
098                    return this;
099            }
100            
101            public Object buildActions(){
102                    System.out.println("creating Actions\n");
103                    modelActions=new ActionGroupImpl(this);
104                    try{
105                            modelActions.createActionForEach$message(bugList,
106                                    new Selector(Class.forName("Bug"),"step",false));
107                            modelActions.createActionTo$message(this,
108                                    new Selector(this.getClass(),"checkToStop",false));
109                    } catch (Exception e) {
110                            e.printStackTrace (System.err);
111                            System.exit(1);
112                    }
113                    
114                    modelSchedule=new ScheduleImpl(this,1);
115                    modelSchedule.at$createAction(0,modelActions);
116                    return this;
117            }
118            
119            public Activity activateIn(Swarm context){
120                    super.activateIn(context);
121            modelSchedule.activateIn(this);
122                    return getActivity();
123            }
124            
125            /**
126             * 餌の数を調べ、食べ尽くされていたらModelSwarmを停止する。
127             */
128            public Object checkToStop(){
129                    
130                    // If the bugs have eaten all the food, the model run is over.
131                    
132                    if(foodSpace.getFood()<=0){
133                            this.getActivity().terminate();
134                            System.out.println("ModelSwarm terminated.");
135                    }
136                    
137                    // if not, increment time and continue running the model
138                    
139                    time++;
140                    return this;
141            }
142            
143            /**
144             * ModelSwarmのスーパークラスであるSwarmImplのインターフェースswarm.defobj.Drop
145             * のメソッド<a href="http://www.santafe.edu/projects/swarm/swarmdocs/refbook-java/swarm/defobj/Drop.html#drop()">
146             * drop<\a>をオーバーロードする。
147             */
148            public void drop(){
149                    foodSpace.drop();
150                    world.drop();
151                    super.drop();
152            }
153    }
