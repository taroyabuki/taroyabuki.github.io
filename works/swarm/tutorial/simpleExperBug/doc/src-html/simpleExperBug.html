001    import swarm.*;
002    
003    /**
004     * これまでは一つのモデルをシミュレートするだけだったのに対し、
005     * ここではパラメータを変えながら複数のモデルをシミュレートする。<BR>
006     * <BR>
007     * 複数のモデルを管理するためにExperSwarmを使う。
008     * これまでのObserverSwarmに相当するクラスである。<BR>
009     * <BR>
010     * <IMG src="../simpleExperBug.png" border="0"><BR>
011     * <BR>
012     * 以上でチュートリアルは完了である。さらに例を見たければ、
013     * jheatbugsやjmousetrapなどがある。Swarmでの開発は、
014     * このチュートリアルのシミュレーションの枠組み（Observerがあって、Modelがある）
015     * にとらわれるものではない。
016     * @author YABUKI Taro
017     * @version 0.4
018     */
019    public class simpleExperBug{
020            public static void main(String[] args) {
021                    ExperSwarm experSwarm;
022                    
023                    Globals.env.initSwarm("bug", "0.4", "YABUKI Taro", args);
024                    
025                    experSwarm=new ExperSwarm(Globals.env.globalZone);
026                    Globals.env.setWindowGeometryRecordName(experSwarm,"experSwarm");
027                    experSwarm.buildObjects();
028                    experSwarm.buildActions();
029                    experSwarm.activateIn(null);
030                    experSwarm.go();
031            }
032    }
