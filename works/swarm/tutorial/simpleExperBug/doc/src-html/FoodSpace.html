001    import swarm.space.*;
002    import swarm.*;
003    import swarm.defobj.*;
004    
005    /**
006     * 餌の数を数えて、食べ尽くされたのがわかるようにする。
007     * Bugは餌を食べると、FoodSpaceのbugAteを呼び出す。
008     * また、ModelSwarmが餌の数を調べられるように、getFoodを追加している。
009     */
010    public class FoodSpace extends Discrete2dImpl{
011            int food;
012            
013            public FoodSpace(Zone aZone,int x,int y){
014                    super(aZone,x,y);
015            }
016            
017            /**
018             * @return 餌の数
019             */
020            public int getFood(){
021                    return food;
022            }
023            
024            public Object seedFoodWithProb(double seedProb){
025                    int x,y;
026                    int xsize,ysize;
027                    
028                    food=0;
029                    xsize=this.getSizeX();
030                    ysize=this.getSizeY();
031                    
032                    for (y = 0; y < ysize; y++)
033                            for (x = 0; x < xsize; x++)
034                                    if(Globals.env.uniformDblRand.getDoubleWithMin$withMax(0.0,1.0)
035                                            <seedProb){
036                                            this.putValue$atX$Y(1,x,y);
037                                            food++;
038                                    }
039                    return this;
040            }
041            
042            /**
043             * 餌を食べたBugから呼ばれるこのメソッドは記録してある餌の数foodをデクリメントする。
044             */
045            public Object bugAte(){
046                    food--;
047                    return this;
048            }
049    }
