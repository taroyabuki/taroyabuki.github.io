001    import swarm.*;
002    
003    /**
004     * 餌がまかれた2次元グリッド上をBugが歩き回る<BR>
005     * <BR>
006     * BugはFoodSpaceという2次元グリッド上を歩く。
007     * FoodSpaceの各グリッドにはseedProbの確率で餌がまかれている。
008     * Bugは餌を見つけるとそれを食べる。<BR>
009     * <BR>
010     * FoodSpaceもオブジェクトであり、simpleObjCBug2中でnewによって作られる。
011     * 生成されたFoodSpaceには、メソッドseedFoodWithProbによって餌がまかれる。<BR>
012     * <BR>
013     * BugもsimpleObjCBugの時と同様に作られる。
014     * 生成されたBugは、メソッドsetFoodSpaceによって自分のFoodSpaceを知る。
015     * Bugのメイン・メソッドはstepであるが、このメソッドはFoodSpaceにアクセスするように修正されている。<BR>
016     * <BR>
017     * <IMG src="../simpleObjCBug2.png" border="0"><BR>
018     * <BR>
019     * 次はsimpleSwarmBug
020     * @author YABUKI Taro
021     * @version 0.4
022     */
023    
024    public class simpleObjCBug2{
025            public static void main(String[] args) {
026                    int worldXSize = 80;
027                    int worldYSize = 80;
028                    
029                    int xPos = 40;
030                    int yPos = 40;
031                    
032                    double seedProb=0.5; // Density of distribution of "food"
033                    
034                    int i;
035                    Globals.env.initSwarm("bug", "0.4", "YABUKI Taro", args);
036                    
037                    // Create and initialize a "food" space
038                    
039                    FoodSpace foodSpace=new FoodSpace(Globals.env.globalZone,worldXSize,worldYSize);
040                    
041                    foodSpace.seedFoodWithProb(seedProb);
042                    
043                    Bug aBug=new Bug(Globals.env.globalZone);
044                    aBug.setFoodSpace(foodSpace);
045                    aBug.setWorldSizeX$Y(worldXSize,worldYSize);
046                    
047                    aBug.setX$Y(xPos,yPos);
048                    
049                    for(i = 0; i < 100; ++i)
050                            aBug.step();
051            }
052    }
