009    /**
010     * あらゆるSwarmのオブジェクトにアクセス可能なProbeを作成する。
011     * Probeはオブジェクトのフィールドの値を表示・変更したり、
012     * メソッドを呼び出したりするためのGUIである。
013     * 注意点として、Probeで扱えるフィールドやメソッドはpublicなものでなければならない。<BR>
014     * <BR>
015     * ここでは、OberverSwarmとModelSwarmへのProbeが起動時に生成されるようにする。
016     * さらにマウスでクリックすることによってBugへのProbeも生成されるようにする。<BR>
017     * <BR>
018     * ProbeはGlobals.env.createArchivedProbeDisplayで作成することができる。
019     * これで作ったProbeにはpublicなフィールドがすべてのっている。
020     * 表示するフィールドを選びたい場合や、メソッドものせたい場合には次のようにすればよい。<BR>
021     * 1. ProbeMapにフィールドやメソッドを追加する<BR>
022     * 2. ProbeLibraryにProbeMapを登録する<BR>
023     * 3. Globals.env.createArchivedProbeDisplay<BR>
024     * ここではObserverSwarmへのProbeはデフォルトのまま、
025     * ModelSwarmへのProbeはカスタマイズしている。<BR>
026     * <BR>
027     * プログラムは2つのProbeを生成した時点で停止する。
028     * ここで、シミュレーションのパラメータ（ModelSwarmのフィールド）
029     * を修正してからStartボタンを押せば、パラメータファイルを書き換えることなく、
030     * さまざまな設定をシミュレートすることができる
031     * （フィールドの数値を書き換えたらEnterキーを押すこと）。<BR>
032     * <BR>
033     * 次の2点も確認してほしい。<BR>
034     * 1. Probe上の青字のオブジェクト名をクリックすると、publicなメソッドもすべて表示される。<BR>
035     * 2. そこで、右上のボタンをクリックすると、スーパークラスのProbeが生成される。<BR>
036     */
037    public class ObserverSwarm extends GUISwarmImpl{
038            /**
039             * probeで操作できるようにするためにはpublicでなければならない
040             */
041            public int displayFrequency; 
042            
043            ActionGroup displayActions; // schedule data structs
044            Schedule displaySchedule;
045            ModelSwarm modelSwarm; // the Swarm we're observing
046            Colormap colorMap; // allocate colours
047            ZoomRaster worldRaster; // 2d display widget
048            Value2dDisplay foodDisplay; // display the heat
049            Object2dDisplay bugDisplay; // display the heatbugs
050            
051            /**
052             * コンストラクタ。ObserverSwarmのProbeはデフォルトのまま
053             * （すべてのpublicなフィールドが表示される）でよいため、独自のprobeMapを作る必要はない。
054             * デフォルトのProbeを再現するようなコードをコメント中に示してあるから参考にしてほしい。
055             * Probeの作り方は、ModelSwarmのコンストラクタを見た方がわかりやすいだろう。
056             */
057            public ObserverSwarm(Zone aZone){
058                    super(aZone);
059                    displayFrequency=1;
060

065

067              ("displayFrequency",this.getClass()));
068            
069            // Now install our custom probeMap into the probeLibrary.
070            
071                    Globals.env.probeLibrary.setProbeMap$For(probeMap,this.getClass());

073            
074            public Object buildObjects(){
075                    super.buildObjects();
076                    
077                    modelSwarm = (ModelSwarm)
078                            Globals.env.lispAppArchiver.getWithZone$key(Globals.env.globalZone,"modelSwarm"); //エラーだった場合の次の処理は未実装
079                    if(modelSwarm==null){
080                            System.out.println("Can't find the modelSwarm parameters.");
081                            System.exit(1);
082                    }
083                    
084                    // Now create probe objects on the model and ourselves. This gives a
085                    // simple user interface to let the user change parameters.
086                    
087                    Globals.env.createArchivedProbeDisplay (modelSwarm, "modelSwarm");
088                    Globals.env.createArchivedProbeDisplay (this, "observerSwarm");
089                    
090                    // Instruct the control panel to wait for a button event.
091                    // We halt here until someone hits a control panel button.
092                    
093                    // Now that we're using Probes, the user can set the parameters
094                    // in the ModelSwarm probe window - we halt here to allow
095                    // the user to change parameters.
096                    
097                    getControlPanel().setStateStopped();
098                    
099                    modelSwarm.buildObjects();
100                    
101                    colorMap=new ColormapImpl(this);
102                    colorMap.setColor$ToName((byte)0,"black");//キャストが必要
103                    colorMap.setColor$ToName((byte)1,"red");
104                    colorMap.setColor$ToName((byte)2,"green");
105                    
106                    worldRaster=new ZoomRasterImpl(this);
107                    worldRaster.setColormap(colorMap);
108                    worldRaster.setZoomFactor(4);
109                    worldRaster.setWidth$Height(
110                            modelSwarm.getWorld().getSizeX(),
111                            modelSwarm.getWorld().getSizeY());
112                    worldRaster.setWindowTitle("Food Space");
113                    worldRaster.pack(); // draw the window.
114                    
115                    foodDisplay=new Value2dDisplayImpl(this,worldRaster,colorMap,modelSwarm.getFood());
116                    
117                    try {
118                            bugDisplay = new Object2dDisplayImpl(
119                                    this,
120                                    worldRaster,
121                                    modelSwarm.getWorld(),
122                                    new Selector(Class.forName("Bug"), "drawSelfOn", false));
123                    } catch (Exception e) {
124                            System.out.println ("Exception: " + e.getMessage ());
125                            System.exit(1);
126                    }
127                    
128                    bugDisplay.setObjectCollection(modelSwarm.getBugList());
129                    
130                    // worldRasterの上でマウスがクリックされた場合、
131                    // bugDisplayのメソッドmakeProbeAtX$Yを呼び出す。
132                    // makeProbeAtX$Yはそこにあるオブジェクト（つまりBug）のProbeを生成する。
133                    
134                    try {
135                            worldRaster.setButton$Client$Message(
136                                    3,bugDisplay,new Selector(bugDisplay.getClass(), "makeProbeAtX$Y",true));
137            } catch (Exception e) {
138                            System.out.println ("Exception: " + e.getMessage ());
139                            System.exit(1);
140                    }
141            
142                    return this;
143            }
144            
145            public Object buildActions(){
146                    super.buildActions();
147                    modelSwarm.buildActions();
148                    displayActions=new ActionGroupImpl(this);
149                    
150                    try {
151                            displayActions.createActionTo$message(foodDisplay,
152                                    new Selector(foodDisplay.getClass(),"display",false));
153                            displayActions.createActionTo$message(bugDisplay,
154                                    new Selector(bugDisplay.getClass(),"display",false));
155                            displayActions.createActionTo$message(worldRaster,
156                                    new Selector(worldRaster.getClass(),"drawSelf",false));
157                            displayActions.createActionTo$message(
158                                    getActionCache(),
159                                    new Selector(getActionCache().getClass(),"doTkEvents",false));
160                    } catch (Exception e) {
161                            System.out.println ("Exception: " + e.getMessage ());
162                            System.exit(1);
163                    }
164                    
165                    displaySchedule = new ScheduleImpl(this,displayFrequency);
166                    displaySchedule.at$createAction(0,displayActions);
167                    
168                    return this;
169        }
170            
171            public Activity activateIn(Swarm context){
172            super.activateIn(context);
173                    modelSwarm.activateIn(this);
174                    displaySchedule.activateIn(this);
175                    return getActivity();
176            }
