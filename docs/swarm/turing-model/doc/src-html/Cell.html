001    import swarm.*;
002    import swarm.objectbase.*;
003    import swarm.defobj.*;
004    
005    public class Cell extends SwarmObjectImpl {
006            double u,v,u1,u2,v1,v2;
007            
008            int ru,rv;
009            double w1,w2,m0,m1,p,d,e;
010            
011            int offset;
012            PatternSpace patternSpace;
013            
014            public Cell(Zone aZone,int Offset,PatternSpace ps){
015                    super(aZone);
016                    offset=Offset;
017                    patternSpace=ps;
018            }
019            
020            public double getU(){
021                    return u;
022            }
023            
024            public double getV2(){
025                    return v2;
026            }
027    
028            public void activate(){
029                    u=1;
030                    initialize();
031            }
032            
033            public void inhibit(){
034                    u=0;
035                    initialize();
036            }
037            
038            void initialize(){
039                    v=0;
040                    u1=0;
041                    u2=0;
042                    v1=0;
043                    v2=0;
044            }
045            
046            public void setParams(int Ru,int Rv,
047                    double W1,double W2,double M0,double M1,double P,double D,double E){
048                    ru=Ru;
049                    rv=Rv;
050                    w1=W1;
051                    w2=W2;
052                    m0=M0;
053                    m1=M1;
054                    p=P;
055                    d=D;
056                    e=E;
057            }
058            
059            public void step1(){
060                    if(v>=1) v1=java.lang.Math.round(v*(1-d)-e);
061                    else v1=0;
062            }
063                    
064            public void step2(){
065                    if(u==0){
066                            if(Globals.env.uniformDblRand.getDoubleWithMin$withMax(0.0,1.0)<p)
067                                    u1=1;
068                            else u1=0;// Ç±ÇÃïîï™ÅAò_ï∂Ç…èëÇ¢ÇƒÇ»Ç¢ÇØÇ«ëÂéñ
069                    } else u1=u;
070            }
071                    
072            public void step3(){
073                    if(u1==1) v2=v1+w1;
074                    else v2=v1;
075            }
076                    
077            public void step4(){
078                    int nu=0;
079                    for(int i=-ru;i<=ru;++i){
080                            if(patternSpace.getUAt(offset+i)==1) nu++;
081                    }
082                    if(u1==0 && nu>java.lang.Math.round(m0+m1*v2)) u2=1;
083                    else u2=u1;
084            }
085            
086            public void step5(){
087                    double meanV2=0;
088                    for(int i=-rv;i<=rv;++i){
089                            meanV2+=patternSpace.getV2At(offset+i);
090                    }
091                    meanV2/=(2.*rv+1.);
092                    v=(int)java.lang.Math.round(meanV2);
093            }
094            
095            public void step6(){
096                    if(v>=w2)
097                            u=0;
098                    else
099                            u=u2;
100            }
101    }
