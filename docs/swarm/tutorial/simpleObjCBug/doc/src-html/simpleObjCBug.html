001    import swarm.*;
002    
003    /**
004     * オブジェクト指向による記述<BR>
005     * <BR>
006     * simpleCBugをオブジェクト指向で書き直すとこのようになる。
007     * simpleObjCBugのObjCはObjective-Cのこと（C言語をオブジェクト言語として拡張したのがObjective-C。
008     * ここで用いているのはJavaなのだが、古い命名規則を残している）。
009     * オブジェクトBugを定義するのは、Bug.javaである。<BR>
010     * <BR>
011     * main methodにはオブジェクトBugを生成するための文new Bugがある。
012     * これは、クラスBugのインスタンス（オブジェクト）を生成するためのものである。<BR>
013     * <BR>
014     * Bugのメソッド呼び出し<BR>
015     *              aBug.setX$Y(xPos,yPos);<BR>
016     * は、Bugのフィールド（内部変数）を設定するためのものである。<BR>
017     * <BR>
018     * <IMG src="../simpleObjCBug.png" border="0"><BR>
019     * <BR>
020     * 次はsimpleObjCBug2
021     * @author YABUKI Taro
022     * @version 0.4
023     */
024    
025    public class simpleObjCBug{
026            public static void main(String[] args) {
027                    int worldXSize = 80;
028                    int worldYSize = 80;
029                    
030                    int xPos = 40;
031                    int yPos = 40;
032                    
033                    int i;
034                    Globals.env.initSwarm("bug", "0.4", "YABUKI Taro", args);
035                    
036                    // Make us aBug please!
037                    Bug aBug=new Bug(Globals.env.globalZone);
038                    
039                    // and initialize it
040                    aBug.setX$Y(xPos,yPos);
041                    aBug.setWorldSizeX$Y(worldXSize,worldYSize);
042                    
043                    for(i = 0; i < 100; ++i)
044                            aBug.step(); // Tell it to act
045            }
046    }
