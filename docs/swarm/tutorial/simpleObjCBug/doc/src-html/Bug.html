001    import swarm.*;
002    import swarm.objectbase.*;
003    import swarm.defobj.*;
004    
005    /**
006     * クラスはその名前のついた1つのjavaファイルの中で定義される（ここではBug.java）<BR>
007     */
008    
009    public class Bug extends SwarmObjectImpl {
010            int xPos, yPos;
011            int worldXSize, worldYSize;
012            
013            /**
014             * @param aZone SwarmのオブジェクトはZoneに生成される
015             */
016            public Bug(Zone aZone){
017                    super(aZone);
018            }
019            
020            /**
021             * 位置を設定するメソッド
022             * @param x x座標
023             * @param y y座標
024             */
025            public Object setX$Y(int x, int y){
026                    xPos = x;
027                    yPos = y;
028                    System.out.println("I started at X = " + xPos + " Y = " + yPos + "\n");
029                    return this;
030            }
031            
032            /**
033             * Bugが住む世界のサイズを設定するメソッド。本来Bugがこれを知る必要はないはず
034             * @param xSize x方向のサイズ
035             * @param ySize y方向のサイズ
036             */
037            public Object setWorldSizeX$Y(int xSize, int ySize){
038                    worldXSize=xSize;
039                    worldYSize=ySize;
040                    return this;
041            }
042            
043            /**
044             * ランダムウォークする
045             */
046            public Object step(){
047                    xPos = xPos + Globals.env.uniformIntRand.getIntegerWithMin$withMax(-1,1);
048                    yPos = yPos + Globals.env.uniformIntRand.getIntegerWithMin$withMax(-1,1);
049                    
050                    xPos = (xPos + worldXSize) % worldXSize;
051                    yPos = (yPos + worldYSize) % worldYSize;
052                    
053                    System.out.println("I moved to X = " + xPos + " Y = " + yPos);
054                    return this;
055            }
056    }
