005    /**
006     * FoodSpaceの実装は簡単であるが、それはSwarmライブラリの一部Discrete2dimplを継承しているためである。
007     * 継承は、次の1行による。<BR>
008     * <BR>
009     * public class FoodSpace extends Discrete2dImpl<BR>
010     * <BR>
011     * これによって、FoodSpaceはDiscrete2dImplが持つすべてのフィールドとメソッドを備えることになる。<BR>
012     * <BR>
013     * Discrete2dimplは格子であり、各セルには整数が格納される。
014     * このオブジェクトには、サイズをセットしたり、セルの値を読み書きするためにメソッドが備わっている。
015     * 詳しくはAPIリファレンスを参照してほしい。<BR>
016     * <BR>
017     * FoodSpaceはDiscrete2dimplに新しいメソッドseedFoodWithProbを加えたものである。
018     * これは各セルの値を確率で1にするというものである。
019     * 1という値は餌があることを意味する。セルのデフォルト値は0で、Bugが餌を食べた後も0になる。
020     */
021    
022    public class FoodSpace extends Discrete2dImpl{
023            public FoodSpace(Zone aZone,int x,int y){
024                    super(aZone,x,y);
025            }
026            
027            public Object seedFoodWithProb(double seedProb){
028                    int x,y;
029                    int xsize,ysize;
030                    
031                    xsize=this.getSizeX();
032                    ysize=this.getSizeY();
033                    
034                    for (y = 0; y < ysize; y++)
035                            for (x = 0; x < xsize; x++)
036                                    if(Globals.env.uniformDblRand.getDoubleWithMin$withMax(0.0,1.0)<seedProb)
037                                            this.putValue$atX$Y(1,x,y);
038                    return this;
039            }
040    }
